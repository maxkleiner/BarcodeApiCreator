function APIBarcode2(AURL, atype, url_name, afile: string): boolean;
var httpq: THttpConnectionWinInet;
    rets: TMemoryStream;  
    heads: TStrings; iht: IHttpConnection; //losthost:THTTPConnectionLostEvent;
begin
  httpq:= THttpConnectionWinInet.Create(true); 
  rets:= TMemoryStream.create;
  //heads:= TStringlist.create;     
  try 
    //heads.add('Accept=image/jpg');
    writ(Format(AURL,[urlencode(atype),urlencode(url_name)]))
    //iht:= httpq.setHeaders(heads);
    httpq.Get(Format(AURL,[urlencode(atype),urlencode(url_name)]), rets);
    if httpq.getresponsecode=200 Then begin
       rets.Position:= 0;
        // Add the '.png' extension to the FileName
       //FileName := ChangeFileExt(FileName, '.png');
       rets.savetofile(afile);
       openfile(afile);
       result:= true;
     end  
      else result:='Failed:'+
             itoa(Httpq.getresponsecode)+Httpq.GetResponseHeader('message');
  except  
    writeln('EWI_HTTP: '+ExceptiontoString(exceptiontype,exceptionparam));
    result:= false;
  finally
    httpq:= Nil;
    //heads.Free;
    rets.Free;
  end;                  
end;                 //}

function APIBarcode3(AURL,atype,url_name,afile: string): boolean;
var encodedURL:String; mapStream: TMemoryStream;
begin
 encodedURL:= Format(AURL,[urlencode(atype),urlencode(url_name)]);
  mapStream:= TMemoryStream.create;
  try
    Wininet_HttpGet(EncodedURL, mapStream);   //WinInet
    mapStream.Position:= 0;
    mapStream.Savetofile(afile); // OpenDoc(apath);
    result:= true;
    S_ShellExecute(afile,'',seCmdOpen);
  except  
    writeln('HTTPS: '+ExceptiontoString(exceptiontype,exceptionparam));
    result:= false;  
  finally
    mapStream.Free;
  end;
end;

procedure SIRegister_Jsons(CL: TPSPascalCompiler);
begin
  CL.AddTypeS('TJsonValueType', '( jvNone, jvNull, jvString, jvNumber, jvBoolean, jvObject, jvArray )');
  CL.AddTypeS('TJsonStructType', '( jsNone, jsArray, jsObject )');
  CL.AddTypeS('TJsonNull', '( jsnull2 )');
  CL.AddTypeS('TJsonEmpty', '( jsempty )');
  SIRegister_TJsonBase(CL);
  CL.AddClassN(CL.FindClass('TOBJECT'),'TJsonObject2');
  CL.AddClassN(CL.FindClass('TOBJECT'),'TJsonArray2');
  SIRegister_TJsonValue(CL);
  SIRegister_TJsonArray2(CL);
  SIRegister_TJsonPair(CL);
  SIRegister_TJsonObject2(CL);
  SIRegister_TJson(CL);
end;
