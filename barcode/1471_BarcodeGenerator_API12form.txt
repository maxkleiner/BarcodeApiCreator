function APIBarcode2(AURL, atype, url_name, afile: string): boolean;
var httpq: THttpConnectionWinInet;
    rets: TMemoryStream;  
    heads: TStrings; iht: IHttpConnection; //losthost:THTTPConnectionLostEvent;
begin
  httpq:= THttpConnectionWinInet.Create(true); 
  rets:= TMemoryStream.create;
  //heads:= TStringlist.create;     
  try 
    //heads.add('Accept=image/jpg');
    writ(Format(AURL,[urlencode(atype),urlencode(url_name)]))
    //iht:= httpq.setHeaders(heads);
    httpq.Get(Format(AURL,[urlencode(atype),urlencode(url_name)]), rets);
    if httpq.getresponsecode=200 Then begin
       rets.Position:= 0;
        // Add the '.png' extension to the FileName
       //FileName := ChangeFileExt(FileName, '.png');
       rets.savetofile(afile);
       openfile(afile);
       result:= true;
     end  
      else result:='Failed:'+
             itoa(Httpq.getresponsecode)+Httpq.GetResponseHeader('message');
  except  
    writeln('EWI_HTTP: '+ExceptiontoString(exceptiontype,exceptionparam));
    result:= false;
  finally
    httpq:= Nil;
    //heads.Free;
    rets.Free;
  end;                  
end;                 //}
